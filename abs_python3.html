AtCoder Beginners Selection(Python3)

ここでは, AtCoder Beginners SelectionをPython3を使って解いたときのソースコードを載せます. D言語の方で概要など説明していますので, こちらではソースコードくらいしか載せません.

  1. Product 
    
a, b = list(map(int, input().split()))

if (a * b) % 2 == 0:
    print("Even")
else:
    print("Odd")
  2. Placing Marbles 
    
# input() はchar配列. count(x)は配列にxが何個あるかを数える
print(input().count('1'))
  3. Shift Only 
    
n = int(input())
a = list(map(int, input().split()))
ans = 1000000007

for i in a:
    tmp = 0
    while (i % 2 == 0 and i > 0):
        tmp += 1
        i /= 2
    
    ans = min(ans, tmp)

print(ans)
  4. Coins 
    
a = int(input())
b = int(input())
c = int(input())
x = int(input())

ans = 0

for i in range(0, a + 1):
    for j in range(0, b + 1):
        for k in range(0, c + 1):
            sum = 500 * i + 100 * j + 50 * k

            if sum == x:
                ans += 1
            elif sum > x:
                break;


print(ans)
  5. Some Sums 
    
n, a, b = list(map(int, input().split()))
ans = 0

for i in range(1, n + 1):
    # 割とスッキリ書けた
    digit_sum = sum(map(int, list(str(i))))

    if a <= digit_sum and digit_sum <= b:
        ans += i

print(ans)
  6. Card Game for Two 
    
n = int(input())
a = list(map(int, input().split()))

# リストの中身を降順ソートする
# 元のリストを保持したいならsorted()
a.sort(reverse = True)

alice, bob = 0, 0

for i in range(0, len(a)):
    if i % 2 == 0:
        alice += a[i]
    else:
        bob += a[i]

print(alice - bob)
  7. Kagami Mochi 
    
n = int(input())

ans = 0
dicts = {} # 空の辞書(連想配列)を生成

for i in range(0, n):
    d = int(input())

    if not(d in dicts):
        dicts[d] = True
        ans += 1

print(ans)
  8. Otoshidama 
    
n, y = list(map(int, input().split()))

flag = False

for i in range(0, n + 1):
    for j in range(0, n + 1 - i):
        k = n - i - j
        money = 10000 * i + 5000 * j + 1000 * k

        if money == y:
            print(i, j, k)
            flag = True # 多重ループ抜ける用
            break

    if flag:
        break


if not(flag):
    print(-1, -1, -1)
  9. Daydream 
    
s = input()

p = len(s)

while p > 0:
    if p >= 7 and s[(p - 7):p] == "dreamer":
        p -= 7
    elif p >= 6 and s[(p - 6):p] == "eraser":
        p -= 6
    elif p >= 5 and s[(p - 5):p] == "dream":
        p -= 5
    elif p >= 5 and s[(p - 5):p] == "erase":
        p -= 5
    else:
        p = -1
        break

if p == 0:
    print("YES")
else:
    print("NO")
  10. Traveling 

    
n = int(input())
st, sx, sy = 0, 0, 0

is_reach = True

for i in range(0, n):
    gt, gx, gy = list(map(int, input().split()))
    t = gt - sy
    dist = abs(gx - sx) + abs(gy - sy)

    if not(t >= dist and (t - dist) % 2 == 0):
        is_reach = False
        break


if is_reach:
    print("Yes")
else:
    print("No")
